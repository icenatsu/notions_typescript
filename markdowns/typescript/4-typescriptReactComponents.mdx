### children

<Pre>
```tsx
import { ReactNode } from "react"

type AvatarProps = {
  username: string;
  avatarUrl: string;
  children: ReactNode
};

const Avatar = ({username, avatarUrl, children}: AvatarProps) => {
}
```
</Pre>

### PropsWithChildren

<Pre>
```tsx
import { PropsWithChildren } from 'react';

type AvatarProps = {
  username: string;
  avatarUrl: string;
};

const Avatar = ({ username, avatarUrl, children}: PropsWithChildren<AvatarProps>) => {
}
```
</Pre>

### ComponentPropsWithoutRef

<Pre>
```ts
type ButtonProps = {
  disabled?: boolean;
  className?: string;
  onClick?: (event: MouseEvent) => void;
  // ... les 300 autres propriétés
}

// ComponentPropsWithoutRef<HTMLElement> rajoute toutes les propriétés propres à l'élément, il rajoute également children
type ButtonProps = {
  disabled?: boolean;
  className?: string;
} & ComponentPropsWithoutRef<"button">
```
</Pre>
